# Оптимизированный образ для inference API

FROM python:3.10-slim as base

# Установка только runtime зависимостей
RUN apt-get update && apt-get install -y \
    libgomp1 \
    && rm -rf /var/lib/apt/lists/*

WORKDIR /app

# Копирование минимальных зависимостей для inference
COPY requirements-inference.txt .
RUN pip install --no-cache-dir -r requirements-inference.txt

# Копирование только необходимых модулей
COPY src/deployment/ ./src/deployment/
COPY src/data/preprocessing.py ./src/data/preprocessing.py
COPY src/models/lstm_model.py ./src/models/lstm_model.py

# Копирование обученной модели и scaler
COPY models/model_quantized.onnx ./models/
COPY models/scaler.pkl ./models/

# API сервер
COPY api/ ./api/

# Создание non-root пользователя
RUN useradd -m -u 1000 apiuser && \
    chown -R apiuser:apiuser /app
USER apiuser

# Healthcheck
HEALTHCHECK --interval=30s --timeout=3s --start-period=40s --retries=3 \
    CMD python -c "import requests; requests.get('http://localhost:8000/health')"

EXPOSE 8000

CMD ["uvicorn", "api.main:app", "--host", "0.0.0.0", "--port", "8000", "--workers", "4"]
